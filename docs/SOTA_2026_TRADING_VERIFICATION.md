async def place_real_order(...):
    """Execute REAL trade via CCXT"""
    
    # 1. Get exchange connector
    connector = self._get_connector(exchange_name)
    
    # 2. Execute order
    if order_type == OrderType.MARKET:
        order = await connector.create_market_order(
            symbol=symbol,
            side="buy" if side == OrderSide.BUY else "sell",
            amount=amount
        )
    else:  # LIMIT
        order = await connector.create_limit_order(
            symbol=symbol,
            side="buy" if side == OrderSide.BUY else "sell",
            amount=amount,
            price=price
        )
    
    # 3. Publish order events
    self.event_bus.publish('real_order.placed', order_data)
    
    # 4. For market orders, also publish fill event
    if order_type == OrderType.MARKET:
        self.event_bus.publish('trading.order_filled', fill_data)
